USER LOGIN if(action == 'getUser'){ const users = sheet.getSheetByName('Users').getDataRange().getValues(); const user = users.find(r => r[3] == e.parameter.email && r[4] == e.parameter.pass); if(!user) return jsonOutput({error:'Invalid email or password'}); return jsonOutput({ id: user[0], name: user[2], email: user[3], phone: user[5], wphone: user[6], wname: user[7], status: user[8], win_amount: parseInt(user[9] || 0), total_won: parseInt(user[10] || 0) }); } // 2. GET ALL USERS FOR ADMIN if(action == 'getAll'){ const users = sheet.getSheetByName('Users').getDataRange().getValues(); const data = users.slice(1).filter(r => r[0]!= "").map(r => ({ id: r[0], date: formatDate(r[1]), name: r[2], email: r[3], phone: r[5], wphone: r[6], wname: r[7], status: r[8], win_amount: parseInt(r[9] || 0), total_won: parseInt(r[10] || 0) })); return jsonOutput(data); } // 3. GET USER HISTORY if(action == 'getHistory'){ const hist = sheet.getSheetByName('History').getDataRange().getValues(); const data = hist.slice(1) .filter(r => r[1] == e.parameter.id) .map(r => ({date: formatDate(r[0]), result: r[3], amount: parseInt(r[4] || 0)})) .reverse(); return jsonOutput(data); } // 4. GET ALL WITHDRAWALS FOR ADMIN if(action == 'getWithdrawals'){ const withdraws = sheet.getSheetByName('Withdrawals').getDataRange().getValues(); const data = withdraws.slice(1).filter(r => r[0]!= "").map(r => ({ id: r[0].getTime(), // use timestamp as ID date: formatDate(r[0]), email: r[2], amount: parseInt(r[3] || 0), wphone: r[4], wname: r[5], status: r[6] })).reverse(); return jsonOutput(data); } return jsonOutput({error:'Invalid action'}); } catch(err) { return jsonOutput({error: err.toString()}); } } function doPost(e) { try { const data = JSON.parse(e.postData.contents); const sheet = SpreadsheetApp.openById(SHEET_ID); const id = new Date().getTime(); // unique ID // 1. NEW USER SIGNUP if(data.action == 'signup'){ sheet.getSheetByName('Users').appendRow([ id, new Date(), data.name, data.email, data.password, data.phone, data.wphone, data.wname, 'Pending', 0, 0 ]); MailApp.sendEmail(ADMIN_EMAIL, 'New Jada Card Signup', `New User: ${data.name}\nEmail: ${data.email}\nPhone: ${data.phone}\nWithdraw To: ${data.wname} - ${data.wphone}`); return jsonOutput({success: true}); } // 2. ADMIN ISSUES CARD if(data.action == 'issue'){ updateUser(data.id, 8, 'Issued'); // Column I = status return jsonOutput({success: true}); } // 3. ADMIN SETS WIN AMOUNT if(data.action == 'setWin'){ updateUser(data.id, 9, parseInt(data.amount)); // Column J = win_amount return jsonOutput({success: true}); } // 4. USER SCRATCHES CARD if(data.action == 'recordWin'){ sheet.getSheetByName('History').appendRow([new Date(), data.id, '', data.result, data.amount]); if(data.result == 'WIN' && data.amount > 0){ const userRow = findRow(data.id); const usersSheet = sheet.getSheetByName('Users'); const currentTotal = parseInt(usersSheet.getRange(userRow, 11).getValue() || 0); usersSheet.getRange(userRow, 11).setValue(currentTotal + parseInt(data.amount)); // Add to total_won usersSheet.getRange(userRow, 9).setValue('Pending'); // Reset status usersSheet.getRange(userRow, 10).setValue(0); // Reset win_amount } return jsonOutput({success: true}); } // 5. USER REQUESTS WITHDRAWAL if(data.action == 'withdraw'){ sheet.getSheetByName('Withdrawals').appendRow([ new Date(), id, data.email, data.amount, data.wphone, data.wname, 'Pending' ]); MailApp.sendEmail(ADMIN_EMAIL, 'New Withdrawal Request - Jada Card', `Admin: ${ADMIN_NAME}\nPhone: ${ADMIN_PHONE}\n\n` + `USER: ${data.email}\nAmount: ${data.amount} UGX\n` + `Withdraw To: ${data.wname}\nMM Number: ${data.wphone}`); return jsonOutput({success: true}); } // 6. ADMIN MARKS WITHDRAWAL AS PAID if(data.action == 'markPaid'){ const withdrawSheet = sheet.getSheetByName('Withdrawals'); const rows = withdrawSheet.getDataRange().getValues(); const rowIndex = rows.findIndex(r => r[0].getTime() == data.id) + 1; if(rowIndex > 0){ withdrawSheet.getRange(rowIndex, 7).setValue('Paid'); // Column G = status const userEmail = rows[rowIndex-1][2]; const amount = rows[rowIndex-1][3]; MailApp.sendEmail(userEmail, 'Jada Card Withdrawal Paid', `Hello,\n\nYour withdrawal request of ${amount} UGX has been PAID to your MM: ${rows[rowIndex-1][5]} - ${rows[rowIndex-1][4]}\n\nThank you for using Jada Card.\nAdmin: ${ADMIN_NAME}`); } return jsonOutput({success: true}); } return jsonOutput({error:'Invalid action'}); } catch(err) { return jsonOutput({error: err.toString()}); } } // HELPER FUNCTIONS function updateUser(id, col, value){ const sheet = SpreadsheetApp.openById(SHEET_ID).getSheetByName('Users'); const row = findRow(id); if(row > 0) sheet.getRange(row, col+1).setValue(value); } function findRow(id){ const sheet = SpreadsheetApp.openById(SHEET_ID).getSheetByName('Users'); const ids = sheet.getRange(1,1,sheet.getLastRow(),1).getValues(); return ids.findIndex(r => r[0] == id) + 1; } function jsonOutput(obj){ return ContentService.createTextOutput(JSON.stringify(obj)).setMimeType(ContentService.MimeType.JSON); } function formatDate(date){ if(!date) return ""; return Utilities.formatDate(new Date(date), "Africa/Kampala", "dd/MM/yyyy HH:mm"); }
